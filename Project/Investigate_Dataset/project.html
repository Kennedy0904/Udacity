Q6. What is the number of movies in each category?¶

In [409]:
# Define function to get all occurance in a column with values split with "|"
def getOccurances(column):
    listOcc = []
    for i in df[column]:
        split = list(map(str, i.split('|')))
        for j in split:
            if j not in listOcc:
                listOcc.append(j)
    return listOcc
In [430]:
# Define function to count for each word in occurance list
def getOccurancesCount(occurancesList, column):
    count = 0
    word_count = dict()
    for i in occurancesList:
        for x in df[column]:
            if i in x.split('|'):
                count+=1;
        word_count[i] = count
        count = 0
    return {k: v for k, v in sorted(word_count.items(), key=lambda x: x[1],reverse=True)}
